ls (list) - wypisz zawartość katalogu
cd NAME (change directory) - otwórz katalogu
cd .. - wróć do katalogu wyżej
pwd - nazwa bierzącego katalogu
mkdir NAME - utwórz katalogu
touch NAME - utwórz plik
cat NAME - wypisz zawartość pliku
echo "TEXT" >> NAME - wpisz TEXT do pliku NAME\
rm NAME - usuń plik
